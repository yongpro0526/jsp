계정 생성 하는 방법

-- 스프링 계정 생성
CREATE USER SPRING IDENTIFIED BY SPRING;

-- 권한 부여
GRANT CONNECT, RESOURCE TO SPRING;

-- 테이블 스페이스 연결
ALTER USER SPRING DEFAULT TABLESPACE USERS QUOTA UNLIMITED ON USERS;

---

테이블 만드는 방법

CREATE TABLE BOOK (
    ID NUMBER(10),
    TITLE VARCHAR2(50),
    AUTHOR VARCHAR2(50)
);

ALTER TABLE BOOK ADD CONSTRAINT BOOK_ID_PK PRIMARY KEY (ID);

CREATE SEQUENCE BOOK_SEQ NOCACHE;
